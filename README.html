"""

def df(x):  
"""  
.  
 
,  
1 .  
:param x: (float) .  
:return: tanh (float).  
"""  
return 1 - np.tanh(x)**2 # tanh

def go_forward(inp):  
"""  
.  
,  
.  
:param inp: ( ) .  
:return: , (float) (numpy array).  
"""  
sum = np.dot(W1, inp) #

sum = np.dot(W2, out) #  
y = f(sum) #

"""

,  
.  
:param epoch: , .  
"""
